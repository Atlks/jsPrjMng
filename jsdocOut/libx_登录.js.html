function 检查登录凭据(凭据) {
    如果(不存在(凭据), 提示并终止(检查结果_凭据不存在))
    如果(长度(凭据.用户名) < 3, 提示并终止(检查结果_凭据无效_用户名格式不对))
    如果(用户名不存在(凭据.用户名), 提示并终止(检查结果_用户名不存在))
    如果(凭据.类型 == "密码", 检查密码(凭据))
    如果(凭据.类型 == "token", 检查token(凭据))

    return __通过;
}



/**
 * 检查cookie凭据  token
 * @param 凭据
 */
function 检查token(凭据) {
    如果(长度(凭据.用户名) < 3, 提示并终止(检查结果_凭据无效_用户名格式不对))

    如果(过期效验(凭据.过期日), 提示并终止(检查结果_凭据过期))
    如果(校验签名不通过(凭据.签名), 提示并终止(检查结果_凭据校验签名不通过))
    如果(校验用户名不通过(凭据.用户名), 提示并终止(检查结果_用户名不存在))
}
